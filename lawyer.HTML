let currentQ = 0;

/* ═══════════════════════════════════════════

   NAVIGATION

function showScreen(id) {

  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));

  const el = document.getElementById(id);

  el.classList.add('active');

  el.scrollTop = 0;

}


function startQuiz() {

  currentQ = 0;

  answers  = [];

  showScreen('s-quiz');

  renderQuestion(0);

}


function restartQuiz() {

  showScreen('s-intro');

}


/* ═══════════════════════════════════════════

   QUIZ LOGIC

═══════════════════════════════════════════ */

function renderQuestion(idx) {

  const q = QUESTIONS[idx];

  const total = QUESTIONS.length;


  // Progress

  document.getElementById('q-count').textContent = `${idx + 1} / ${total}`;

  document.getElementById('prog-bar').style.width = `${((idx) / total) * 100}%`;

  document.getElementById('q-num').textContent = `Q.${String(idx + 1).padStart(2, '0')}`;

  document.getElementById('q-text').textContent = q.q;


  // Options

  const labels = ['A', 'B', 'C'];

  const optsEl = document.getElementById('opts');

  optsEl.innerHTML = '';

  q.opts.forEach((txt, i) => {

    const btn = document.createElement('button');

    btn.className = 'opt';

    btn.innerHTML = `<span class="opt-abc">${labels[i]}</span><span class="opt-text">${txt}</span>`;

    btn.onclick = () => selectOption(i);

    optsEl.appendChild(btn);

  });

}


function selectOption(optIdx) {

  const opts = document.querySelectorAll('.opt');

  opts.forEach((b, i) => {

    b.classList.toggle('selected', i === optIdx);

    b.classList.toggle('dimmed', i !== optIdx);

    b.onclick = null; // prevent double tap

  });


  // flash feedback

  const flash = document.getElementById('flash');

  flash.classList.add('on');

  setTimeout(() => flash.classList.remove('on'), 80);


  answers[currentQ] = optIdx;


  setTimeout(() => {

    if (currentQ < QUESTIONS.length - 1) {

      const wrap = document.getElementById('q-wrap');

      wrap.classList.add('out-left');

      setTimeout(() => {

        currentQ++;

        renderQuestion(currentQ);

        wrap.classList.remove('out-left');

        wrap.style.opacity = '0';

        wrap.style.transform = 'translateX(20px)';

        setTimeout(() => {

          wrap.style.transition = 'opacity 0.25s ease, transform 0.25s ease';

          wrap.style.opacity = '1';

          wrap.style.transform = 'translateX(0)';

        }, 20);

      }, 220);

    } else {

      // Last question

      setTimeout(showResult, 300);

    }

  }, 420);

}


/* ═══════════════════════════════════════════

   SCORING & RESULT

═══════════════════════════════════════════ */

function calcScore() {

  let total = 0;

  QUESTIONS.forEach((q, i) => {

    const opt = answers[i] ?? 1; // default mid

    const pts = q.fwd ? (opt + 1) : (3 - opt);

    total += pts;

  });

  return total;

}


function getType(score) {

  // Hidden TRAP: Q11 = C (idx 2) AND Q1 = A (idx 0)

  if (answers[10] === 2 && answers[0] === 0) return TRAP_TYPE;


  for (const t of TYPES) {

    if (score >= t.min && score <= t.max) return t;

  }

  return TYPES[TYPES.length - 1];

}
